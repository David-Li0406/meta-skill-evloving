# SQL Patterns

## Schema for SQL Results

```typescript
import { SqlClient, SqlSchema } from "@effect/sql"

const User = Schema.Struct({
  id: UserId,
  email: Schema.String,
  name: Schema.String,
  created_at: Schema.DateFromSelf,
})

class UserRepo extends Effect.Service<UserRepo>()("UserRepo", {
  dependencies: [SqlClient.SqlClient.Default],
  effect: Effect.gen(function* () {
    const sql = yield* SqlClient.SqlClient

    return {
      findById: Effect.fn("UserRepo.findById")(
        (id: UserId) =>
          sql`SELECT * FROM users WHERE id = ${id}`.pipe(
            SqlSchema.findOne(User)
          )
      ),
    }
  }),
}) {}
```

## SqlSchema Helpers

```typescript
// findOne - returns Option<T>
const user = yield* sql`SELECT * FROM users WHERE id = ${id}`.pipe(
  SqlSchema.findOne(User)
)

// findAll - returns Array<T>
const users = yield* sql`SELECT * FROM users WHERE active = true`.pipe(
  SqlSchema.findAll(User)
)

// single - returns T, fails if not exactly one row
const user = yield* sql`SELECT * FROM users WHERE id = ${id}`.pipe(
  SqlSchema.single(User)
)

// void - for INSERT/UPDATE/DELETE
yield* sql`UPDATE users SET name = ${name} WHERE id = ${id}`.pipe(
  SqlSchema.void
)
```

## Model.Class for Repository Entities

```typescript
import { Model } from "@effect/sql"

class User extends Model.Class<User>("User")({
  id: Model.Generated(UserId),
  email: Schema.String,
  name: Schema.String,
  createdAt: Model.DateTimeInsert,
  updatedAt: Model.DateTimeUpdate,
}) {}

// Model.Generated - auto-generated by DB
// Model.DateTimeInsert - set on insert
// Model.DateTimeUpdate - updated on each update
```

## Repository Pattern

```typescript
class UserRepo extends Effect.Service<UserRepo>()("UserRepo", {
  dependencies: [SqlClient.SqlClient.Default],
  effect: Effect.gen(function* () {
    const sql = yield* SqlClient.SqlClient

    const findById = Effect.fn("UserRepo.findById")(
      (id: UserId): Effect.Effect<Option.Option<User>> =>
        sql`SELECT * FROM users WHERE id = ${id}`.pipe(
          SqlSchema.findOne(User)
        )
    )

    const findByEmail = Effect.fn("UserRepo.findByEmail")(
      (email: string): Effect.Effect<Option.Option<User>> =>
        sql`SELECT * FROM users WHERE email = ${email}`.pipe(
          SqlSchema.findOne(User)
        )
    )

    const create = Effect.fn("UserRepo.create")(
      (input: UserInput): Effect.Effect<User> =>
        sql`INSERT INTO users (email, name) VALUES (${input.email}, ${input.name}) RETURNING *`.pipe(
          SqlSchema.single(User)
        )
    )

    const update = Effect.fn("UserRepo.update")(
      (id: UserId, input: Partial<UserInput>): Effect.Effect<User> =>
        sql`UPDATE users SET
            name = COALESCE(${input.name}, name),
            email = COALESCE(${input.email}, email),
            updated_at = NOW()
          WHERE id = ${id} RETURNING *`.pipe(
          SqlSchema.single(User)
        )
    )

    const delete_ = Effect.fn("UserRepo.delete")(
      (id: UserId): Effect.Effect<void> =>
        sql`DELETE FROM users WHERE id = ${id}`.pipe(
          SqlSchema.void
        )
    )

    return { findById, findByEmail, create, update, delete: delete_ }
  }),
}) {}
```

## Transactions

```typescript
class OrderService extends Effect.Service<OrderService>()("OrderService", {
  dependencies: [SqlClient.SqlClient.Default, UserRepo.Default, OrderRepo.Default],
  effect: Effect.gen(function* () {
    const sql = yield* SqlClient.SqlClient
    const users = yield* UserRepo
    const orders = yield* OrderRepo

    return {
      createOrder: Effect.fn("OrderService.createOrder")(
        (userId: UserId, items: OrderItem[]) =>
          sql.withTransaction(
            Effect.gen(function* () {
              // All operations in same transaction
              const user = yield* users.findById(userId).pipe(
                Effect.flatMap(Option.match({
                  onNone: () => Effect.fail(new UserNotFoundError({ userId })),
                  onSome: Effect.succeed,
                }))
              )

              const order = yield* orders.create({ userId, items })

              yield* orders.createItems(order.id, items)

              return order
            })
          )
      ),
    }
  }),
}) {}
```

## Transaction Isolation Levels

```typescript
sql.withTransaction(effect, {
```

## Batch Operations

```typescript
const createMany = Effect.fn("UserRepo.createMany")(
  (inputs: UserInput[]): Effect.Effect<User[]> =>
    Effect.all(
      inputs.map((input) =>
        sql`INSERT INTO users (email, name) VALUES (${input.email}, ${input.name}) RETURNING *`.pipe(
          SqlSchema.single(User)
        )
      ),
      { concurrency: "unbounded" }
    )
)
```

## Raw Queries with Type Safety

```typescript
// For complex queries, decode manually
const complexQuery = Effect.fn("UserRepo.complexQuery")(
  (filters: Filters) =>
    Effect.gen(function* () {
      const rows = yield* sql`
        SELECT u.*, COUNT(o.id) as order_count
        FROM users u
        LEFT JOIN orders o ON o.user_id = u.id
        WHERE u.active = ${filters.active}
        GROUP BY u.id
        HAVING COUNT(o.id) > ${filters.minOrders}
      `
```
